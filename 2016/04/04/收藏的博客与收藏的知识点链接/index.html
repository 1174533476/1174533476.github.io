//testcode20160403
package com.jellygogo.basejava;
import java.lang.ref.WeakReference;
public class WeakReferenceTest {
public static void main(String[] args) {
WeakReference w = new WeakReference(new WeakString());
StrongString s1 = new StrongString();
StrongString s2 = new StrongString();
int i=0;
while(true){
if(w.get()!=null){
System.out.println(“test”+(i++));
}else{
System.out.println(“end weakReference is NULL”);
break;
}
}
System.out.println(w.get());
System.out.println(s1);
System.out.println(s2);
}
}
class WeakString{
public String s = “ssssssss1”;
public String toString() {
return “s1”;
}
}
class StrongString{
public String s = “ssssssss2”;
public String toString() {
return “s2”;
}
}
